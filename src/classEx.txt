class Animal {
    String name;
    int age;
    function Animal(int age, String name) {
        print(name);
        this.age=age;
        this.name=name;
    };
    function printName() {
        print(this.name);
    };
};
class Dog extends Animal {
    String name;
    function Dog(int age, String name) {
        print(name);
        this.super(age, name);
    };
    function getName() {
        return name;
    };
};
Dog d = new Dog(10, "Tom");
print(d.age);
